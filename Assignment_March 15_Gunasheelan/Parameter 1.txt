package parameter;
 //with parameter and no return type
public class Parameter1 {
    
    public static void add(int x, int y)
    {
        int sum = x + y;
        System.out.print("Sum of two numbers is :" + sum);
    }
    public static void main(String args[])
    {
        int a = 10;
        int b = 20;
 
        add(a, b);
    }
}
